balena_tools.balena_client

BalenaClient is a class that wraps the Balena SDK and BalenaMongoCache to provide a more user-friendly interface to the Balena API.

The BalenaMongoCache is a class that caches data from the Balena API to avoid making unnecessary API calls. The cached data is stored in a local MongoDB database using montydb.

The BalenaClient class requires an authentication token to be set in the BALENA_AUTH_TOKEN environment variable or in a .balena/token file in the user's home directory (this is what is generated with balena login).

When adding additional methods to the BalenaClient class, used the cached data from BalenaMongoCache to avoid making unnecessary API calls, where it makes sense to do so.

Example code

client = BalenaClient(1000)

query = {"device_name":{"$regex": "SVW-CB.*"}}
projection = {"device_name": 1, "uuid": 1, "is_online": 1, "device_tags": 1}
devices = client.get_devices(query, projection)

applications = client.get_applications({"app_name": "Tater_SAI"}, {"app_name": 1, "id": 1, "uuid": 1})

releases = client.get_releases(fleet="FM_HUB_K1", query={"release_tags.version": "v1.0.10"})

 36def auth_token():
 37    """Load the Balena Auth Token from environment or file and returns it. Can be used outside of BalenaClient."""

 39    if token:
 40        return token

279    def _get_release_id_from_identifier(self, fleet: str, release_identifier: str):
280        ''' Convert release commit, id, or version to release id. '''
281        commit_pattern = r'^[0-9a-f]{32}$'
282        id_pattern = r'^\d+$'
283        if re.match(commit_pattern, release_identifier):
284            releases = self.get_releases(fleet, {"commit": release_identifier}, {'id': 1})
285            if len(releases) != 1:
286                raise ValueError(f"Expected 1 release for commit {release}, got {len(releases)}")
287            release_id = releases[0]['id']
288        elif re.match(id_pattern, release_identifier):
289            release_id = release_identifier
290        else:
291            releases = self.get_releases(fleet, {"release_tags.version": release_identifier}, {'id': 1})
292            if len(releases) != 1:
293                raise ValueError(f"Expected 1 release for version {release}, got {len(releases)}")
294            release_id = releases[0]['id']
295
296        return release_id

299    def update_device_to_release(self, device_uuid: str, release_identifier: str):
300        ''' Update devices to a specific release.
301
302        :param device_uuid (str): The uuid of the device to update.
303        :param release_identifier (str): The release to update the device to, can be specified by commit, id, or version (e.g. v1.0.10).
304        '''
305        # Get fleet from the device so we can get the release_id from the release_identifier
306        application_id = self.get_devices({"uuid": device_uuid}, {"belongs_to__application.__id": 1})[0]['belongs_to__application']['__id']
307        fleet = self.get_applications({"id": application_id}, {"app_name": 1})[0]['app_name']
308
309        # Get the release_id from the release_identifier
310        release_id = self._get_release_id_from_identifier(fleet, release_identifier)
311
312        # Get the release id that the device is currently running
313        current_release_id = self.get_devices({"uuid": device_uuid}, {"should_be_running__release.__id": 1})[0]['should_be_running__release']['__id']
314
315        if current_release_id == release_id:
316            return False, f"Already at {release_identifier}"
317
318        # Using the API because the SDK device pinning is not working: self.balena.models.device.pin_to_release(device_uuid, int(release_id))
319        # https://github.com/balena-io/balena-sdk-python/issues/374
320        import requests
321        url = f'https://api.balena-cloud.com/v7/device(uuid=\'{device_uuid}\')'
322        headers = {
323            'Content-Type': 'application/json',
324            'Authorization': f'Bearer {self.auth_token}'
325        }
326        data = {"is_pinned_on__release": int(release_id)}
327        response = requests.patch(url, headers=headers, data=json.dumps(data))
328        if response.status_code == 200:
329            return True, None
330        else:
331            print(f'response.json(): {response.json()}')
332            return False, response.json()

334    def is_device_updated(self, device_uuid: str, release_identifier: str):
335        ''' Check if a device is updated to a specific release.
336
337        :param device_uuid (str): The uuid of the device to check.
338        :param release_identifier (str): The release to check if the device is updated to, can be specified by commit, id, or version (e.g. v1.0.10).
339        '''
340        # Get fleet from the device so we can get the release_id from the release_identifier
341        application_id = self.get_devices({"uuid": device_uuid}, {"belongs_to__application.__id": 1})[0]['belongs_to__application']['__id']
342        fleet = self.get_applications({"id": application_id}, {"app_name": 1})[0]['app_name']
343
344        release_id = self._get_release_id_from_identifier(fleet, release_identifier)
345
346        # Makes sense to not use the cache here because we want to know the current state of the device
347        device = self.balena.models.device.get(device_uuid)
348        return device['is_running__release']['__id'] == int(release_id)

280    def _get_release_id_from_identifier(self, fleet: str, release_identifier: str):
281        ''' Convert release commit, id, or version to release id. '''
282        commit_pattern = r'^[0-9a-f]{32}$'
283        id_pattern = r'^\d+$'
284        if re.match(commit_pattern, release_identifier):
285            releases = self.get_releases(fleet, {"commit": release_identifier}, {'id': 1})
286            if len(releases) != 1:
287                raise ValueError(f"Expected 1 release for commit {release}, got {len(releases)}")
288            release_id = releases[0]['id']
289        elif re.match(id_pattern, release_identifier):
290            release_id = release_identifier
291        else:
292            releases = self.get_releases(fleet, {"release_tags.version": release_identifier}, {'id': 1})
293            if len(releases) != 1:
294                raise ValueError(f"Expected 1 release for version {release}, got {len(releases)}")
295            release_id = releases[0]['id']
296
297        return release_id

302

304        :param release_identifier (str): The release to update the device to, can be specified by commit, id, or version (e.g. v1.0.10).
305        '''
306        # Get fleet from the device so we can get the release_id from the release_identifier
307        application_id = self.get_devices({"uuid": device_uuid}, {"belongs_to__application.__id": 1})[0]['belongs_to__application']['__id']
308        fleet = self.get_applications({"id": application_id}, {"app_name": 1})[0]['app_name']
309
310        # Get the release_id from the release_identifier
311        release_id = self._get_release_id_from_identifier(fleet, release_identifier)
312
313        # Get the release id that the device is currently running
314        current_release_id = self.get_devices({"uuid": device_uuid}, {"should_be_running__release.__id": 1})[0]['should_be_running__release']['__id']
315
316        if current_release_id == release_id:
317            return False, f"Already at {release_identifier}"
318
319        # Using the API because the SDK device pinning is not working: self.balena.models.device.pin_to_release(device_uuid, int(release_id))
320        # https://github.com/balena-io/balena-sdk-python/issues/374
321        import requests
322        url = f'https://api.balena-cloud.com/v7/device(uuid=\'{device_uuid}\')'
323        headers = {
324            'Content-Type': 'application/json',
325            'Authorization': f'Bearer {self.auth_token}'
326        }
327        data = {"is_pinned_on__release": int(release_id)}
328        response = requests.patch(url, headers=headers, data=json.dumps(data))
329        if response.status_code == 200:
330            return True, None
331        else:
332            print(f'response.json(): {response.json()}')
333            return False, response.json()
